100     public static void normalize(final KAFDocument document) {
101 
102         // Convert SST, synset and BBN attributes to external refs
103         for (final Term term : document.getTerms()) {
104             boolean hasBBN = false;
105             boolean hasSynset = false;
106             boolean hasSST = false;
107             for (final ExternalRef ref : term.getExternalRefs()) {
108                 hasBBN |= RESOURCE_BBN.equalsIgnoreCase(ref.getResource());
109                 hasSynset |= RESOURCE_WN_SYNSET.equalsIgnoreCase(ref.getResource());
110                 hasSST |= RESOURCE_WN_SST.equalsIgnoreCase(ref.getResource());
111             }
112             if (!hasBBN && term.getBBNTag() != null) {
113                 term.addExternalRef(document.newExternalRef(RESOURCE_BBN, term.getBBNTag()));
114             }
115             if (!hasSynset && term.getWordnetSense() != null) {
116                 term.addExternalRef(document.newExternalRef(RESOURCE_WN_SYNSET,
117                         term.getWordnetSense()));
118             }
119             if (!hasSST && term.getSupersenseTag() != null) {
120                 term.addExternalRef(document.newExternalRef(RESOURCE_WN_SST,
121                         term.getSupersenseTag()));
122             }
123             term.setBBNTag(null);
124             term.setWordnetSense(null);
125             term.setSupersenseTag(null);
126         }
127 
128         // Remove duplicate external refs
129         for (final Predicate predicate : document.getPredicates()) {
130             normalizeRefs(getRefs(predicate));
131             for (final Role role : predicate.getRoles()) {
132                 normalizeRefs(getRefs(role));
133             }
134         }
135     }

137     public static List<Term> filterTerms(final Iterable<Term> terms) {
138         final List<Term> result = Lists.newArrayList();
139         boolean atBeginning = true;
140         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
141             final char pos = Character.toUpperCase(term.getPos().charAt(0));
142             if (atBeginning && (pos == 'D' || pos == 'P')) {
143                 continue;
144             }
145             for (final WF word : term.getWFs()) {
146                 final String text = word.getForm();
147                 if (SYMBOLS.contains(text) || !WF_EXCLUSION_PATTERN.matcher(text).matches()) {
148                     result.add(term);
149                     atBeginning = false;
150                     break;
151                 }
152             }
153         }
154         return result;
155     }

157     public static String getText(final Iterable<Term> terms) {
158         final StringBuilder builder = new StringBuilder();
159         boolean atBeginning = true;
160         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
161             final boolean properNoun = term.getMorphofeat().startsWith("NNP");
162             for (final WF word : term.getWFs()) {
163                 builder.append(atBeginning ? "" : " ");
164                 builder.append(properNoun ? word.getForm() : word.getForm().toLowerCase());
165                 atBeginning = false;
166             }
167         }
168         return builder.toString();
169     }

185     public static Set<Term> extractHeads(final KAFDocument document,
186             @Nullable final Iterable<Term> ancestors, @Nullable final Iterable<Term> span,
187             @Nullable final java.util.function.Predicate<Term> predicate) {
188 
189         Set<Term> ancestorSet;
190         if (ancestors != null) {
191             ancestorSet = ImmutableSet.copyOf(ancestors);
192         } else {
193             ancestorSet = Sets.newHashSet();
194             final Set<Term> termSet = Sets.newHashSet(span);
195             for (final Term term : termSet) {
196                 final Dep dep = document.getDepToTerm(term);
197                 if (dep == null || !termSet.contains(dep.getFrom())) {
198                     ancestorSet.add(term);
199                 }
200             }
201         }
202 
203         final Set<Term> result = Sets.newHashSet();
204         for (final Term ancestor : ancestorSet) {
205             extractHeadsHelper(document, ancestor, predicate, result);
206         }
207         if (span != null) {
208             result.retainAll(ImmutableSet.copyOf(span));
209         }
210         // System.err.println(document.getPublic().uri + " -> " + termFilter + " / " + ancestors
211         // + " -> " + result);
212         return result;
213     }

217     private static boolean extractHeadsHelper(final KAFDocument document, final Term term,
218             final java.util.function.Predicate<Term> predicate, final Collection<Term> result) {
219         final String pos = extendedPos(document, term);
220         boolean accepted = false;
221         if (pos.startsWith("V")) {
222             final Term srlHead = syntacticToSRLHead(document, term);
223             if (!term.equals(srlHead)) {
224                 accepted = extractHeadsHelper(document, srlHead, predicate, result);
225             }
226         }
227         if (!accepted && (predicate == null || predicate.test(term))) {
228             result.add(term);
229             accepted = true;
230         }
231         if (accepted) {
232             for (final Dep dep : document.getDepsFromTerm(term)) {
233                 if (dep.getRfunc().toUpperCase().contains("COORD")) {
234                     extractHeadsHelper(document, dep.getTo(), predicate, result);
235                 }
236             }
237         } else {
238             for (final Dep dep : document.getDepsFromTerm(term)) {
239                 extractHeadsHelper(document, dep.getTo(), predicate, result);
240             }
241         }
242         return accepted;
243     }

245     public static boolean hasHead(final KAFDocument document, final Object annotation,
246             final Term head) {
247         List<Span<Term>> spans;
248         if (annotation instanceof Coref) {
249             spans = ((Coref) annotation).getSpans();
250         } else if (annotation instanceof Entity) {
251             spans = ((Entity) annotation).getSpans();
252         } else if (annotation instanceof Timex3) {
253             spans = ImmutableList.of(KAFDocument.newTermSpan(document
254                     .getTermsByWFs(((Timex3) annotation).getSpan().getTargets())));
255         } else if (annotation instanceof Predicate) {
256             spans = ImmutableList.of(((Predicate) annotation).getSpan());
257         } else if (annotation instanceof Role) {
258             spans = ImmutableList.of(((Role) annotation).getSpan());
259         } else {
260             throw new IllegalArgumentException("Unsupported annotation: " + annotation);
261         }
262         for (final Span<Term> span : spans) {
263             if (head == extractHead(document, span)) {
264                 return true;
265             }
266         }
267         return false;
268     }

270     public static Span<Term> getNominalSpan(final KAFDocument document, final Term term,
271             final boolean includeCoord, final boolean includeModifiers) {
272 
273         // Start from the supplied term
274         final Set<Term> terms = Sets.newHashSet(term);
275 
276         // Identify head and terms of all NE and TIMEX markables containing supplied term
277         final Map<Term, List<Term>> markables = Maps.newHashMap();
278         for (final Entity entity : document.getEntitiesByTerm(term)) {
279             markables.put(document.getTermsHead(entity.getTerms()), entity.getTerms());
280         }
281         for (final WF wf : term.getWFs()) {
282             for (final Timex3 timex : document.getTimeExsByWF(wf)) {
283                 final List<Term> span = document.getTermsByWFs(timex.getSpan().getTargets());
284                 markables.put(document.getTermsHead(span), span);
285             }
286         }
287 
288         // Add the terms of the smallest markable 'matching' the term (i.e., whose head matches
289         // the term or a term ancestor in the dependency tree)
290         if (!markables.isEmpty()) {
291             Term t = term;
292             while (true) {
293                 final List<Term> parent = markables.get(t);
294                 if (parent != null) {
295                     terms.addAll(parent);
296                     break;
297                 }
298                 final Dep dep = document.getDepToTerm(t);
299                 if (dep == null) {
300                     break;
301                 }
302                 t = dep.getFrom();
303             }
304         }
305 
306         // Identify head
307         final Term head = document.getTermsHead(terms);
308 
309         // Add all terms reachable from the head using a regex
310         final String regex = includeCoord ? includeModifiers ? "(COORD CONJ?)* ((NAME|NMOD|AMOD|TMP) .*)?"
311                 : "(COORD CONJ?)* NAME"
312                 : includeModifiers ? "((NAME|NMOD|AMOD|TMP) .*)?" : "NAME";
313         terms.addAll(document.getTermsByDepAncestors(Collections.singleton(head), regex));
314 
315         // Sort obtained terms by offset and return resulting list
316         return KAFDocument.newTermSpan(Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms),
317                 head);
318     }

387                     break;

390                 }
391             }

410             }
411         }

522     public static List<Range> termRangesFor(final KAFDocument document, final Iterable<Term> terms) {
523         final List<Range> ranges = Lists.newArrayList();
524         int startIndex = -1;
525         int lastIndex = -2;
526         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
527             final int termIndex = document.getTerms().indexOf(term);
528             if (termIndex - lastIndex > 1) {
529                 if (startIndex >= 0) {
530                     ranges.add(Range.create(startIndex, lastIndex + 1));
531                 }
532                 startIndex = termIndex;
533             }
534             lastIndex = termIndex;
535         }
536         if (startIndex != -1 && lastIndex >= startIndex) {
537             ranges.add(Range.create(startIndex, lastIndex + 1));
538         }
539         return ranges;
540     }

542     public static List<Range> rangesFor(final KAFDocument document, final Iterable<Term> terms) {
543         final List<Range> ranges = Lists.newArrayList();
544         int startOffset = -1;
545         int endOffset = -1;
546         int termIndex = -2;
547         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
548             final int lastTermIndex = termIndex;
549             termIndex = document.getTerms().indexOf(term);
550             if (termIndex - lastTermIndex > 1) {
551                 if (startOffset != -1) {
552                     ranges.add(Range.create(startOffset, endOffset));
553                 }
554                 startOffset = term.getOffset();
555             }
556             endOffset = NAFUtils.getEnd(term);
557         }
558         if (startOffset != -1 && endOffset > startOffset) {
559             ranges.add(Range.create(startOffset, endOffset));
560         }
561         return ranges;
562     }

578     @Nullable
579     public static Span<Term> trimSpan(@Nullable final Span<Term> span, final int sentenceID) {
580         if (span == null || span.isEmpty()) {
581             return null;
582         }
583         boolean sameSentence = true;
584         for (final Term term : span.getTargets()) {
585             if (term.getSent() != sentenceID) {
586                 sameSentence = false;
587                 break;
588             }
589         }
590         if (sameSentence) {
591             return span;
592         }
593         final List<Term> filteredTerms = Lists.newArrayList();
594         for (final Term term : span.getTargets()) {
595             if (term.getSent() == sentenceID) {
596                 filteredTerms.add(term);
597             }
598         }
599         final Span<Term> result = KAFDocument.newTermSpan(filteredTerms);
600         for (final Term head : span.getHeads()) {
601             if (head.getSent() == sentenceID) {
602                 result.getHeads().add(head);
603             }
604         }
605         return result;
606     }

610     public static Span<Term> normalizeSpan(final KAFDocument document,
611             @Nullable final Span<Term> span) {
612 
613         // Handle null and empty span
614         if (span == null || Iterables.isEmpty(span.getTargets())) {
615             return KAFDocument.newTermSpan();
616         }
617 
618         // Identify all the 'root' terms in the span whose dep tree parent is outside the span
619         final Set<Term> roots = Sets.newHashSet();
620         final Set<Term> terms = ImmutableSet.copyOf(span.getTargets());
621         for (final Term term : terms) {
622             final Dep dep = document.getDepToTerm(term);
623             if (dep == null || !terms.contains(dep.getFrom())) {
624                 roots.add(term);
625             }
626         }
627 
628         // If only one 'root', return the normalized span having that root as the head
629         if (roots.size() == 1) {
630             return KAFDocument.newTermSpan(span.getTargets(), roots.iterator().next());
631         }
632 
633         // Otherwise, look for the closest head outside the span. First compute all the paths from
634         // the dep tree roots to the 'root' terms identified before
635         final List<List<Term>> paths = Lists.newArrayList();
636         for (final Term root : roots) {
637             final List<Term> path = Lists.newArrayList(root);
638             for (Dep dep = document.getDepToTerm(root); dep != null; dep = document
639                     .getDepToTerm(dep.getFrom())) {
640                 path.add(dep.getFrom());
641             }
642             Collections.reverse(path);
643             paths.add(path);
644         }
645 
646         // Then look for the deepest node common to all those paths
647         int depth = 0;
648         Term externalHead = null;
649         outer: for (; depth < paths.get(0).size(); ++depth) {
650             final Term t = paths.get(0).get(depth);
651             for (int i = 1; i < paths.size(); ++i) {
652                 final List<Term> path = paths.get(i);
653                 if (depth >= path.size() || !path.get(depth).equals(t)) {
654                     break outer;
655                 }
656             }
657             externalHead = t;
658         }
659 
660         // If found, compute the terms for the external span
661         Set<Term> externalTerms = null;
662         if (externalHead != null) {
663             externalTerms = Sets.newHashSet(terms);
664             externalTerms.add(externalHead);
665             for (final List<Term> path : paths) {
666                 externalTerms.addAll(path.subList(depth, path.size()));
667             }
668         }
669 
670         // Now look for the internal head that covers the most part terms of the span. Start by
671         // associating to each candidate internal head the terms it would cover
672         final Multimap<Term, Term> map = HashMultimap.create();
673         for (final Term term : terms) {
674             Dep dep = document.getDepToTerm(term);
675             if (dep == null) {
676                 map.put(term, term);
677             } else {
678                 for (; dep != null; dep = document.getDepToTerm(dep.getFrom())) {
679                     if (!terms.contains(dep.getFrom())) {
680                         map.put(dep.getTo(), term);
681                         break;
682                     }
683                 }
684             }
685         }
686 
687         // Then identify the best internal head
688         Term internalHead = null;
689         Collection<Term> internalTerms = null;
690         for (final Map.Entry<Term, Collection<Term>> entry : map.asMap().entrySet()) {
691             if (internalHead == null || entry.getValue().size() >= internalTerms.size()) {
692                 internalTerms = entry.getValue();
693                 internalHead = entry.getKey();
694             }
695         }
696 
697         // Return either the external span (if defined) or the internal one, based on which one is
698         // most similar in size to the original span (if equal, prefer external one).
699         if (externalTerms != null
700                 && externalTerms.size() - terms.size() <= terms.size() - internalTerms.size()) {
701             return KAFDocument.newTermSpan(
702                     Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(externalTerms), externalHead);
703         } else {
704             return KAFDocument.newTermSpan(
705                     Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(internalTerms), internalHead);
706         }
707     }

709     public static List<Span<Term>> mergeSpans(final KAFDocument document,
710             final Iterable<Span<Term>> spans, final boolean canAddTerms) {
711 
712         // Build a map associating to each span head the other heads it is coordinated with
713         final Map<Term, List<Term>> extents = Maps.newHashMap();
714         final Map<Term, Set<Term>> clusters = Maps.newHashMap();
715         for (final Span<Term> span : spans) {
716             final Term head = extractHead(document, span);
717             clusters.put(head, Sets.newHashSet(head));
718             extents.put(head, span.getTargets());
719         }
720         for (final Term head : clusters.keySet()) {
721             for (Dep dep = document.getDepToTerm(head); dep != null
722                     && ("CONJ".equals(dep.getRfunc()) || "COORD".equals(dep.getRfunc())); dep = document
723                     .getDepToTerm(dep.getFrom())) {
724                 if (clusters.keySet().contains(dep.getFrom())) {
725                     clusters.get(head).add(dep.getFrom());
726                     clusters.get(dep.getFrom()).add(head);
727                 } else if ("CO".indexOf(dep.getFrom().getPos()) < 0) {
728                     break; // don't include intermediate terms that are not conjunctions or commas
729                 }
730             }
731         }
732 
733         // Create a span for each cluster of heads, including intermediate conjunctions
734         final List<Span<Term>> result = Lists.newArrayList();
735         while (!clusters.isEmpty()) {
736             final Set<Term> heads = clusters.values().iterator().next();
737             final Set<Term> terms = Sets.newHashSet();
738             Term spanHead = heads.iterator().next();
739             for (final Term head : heads) {
740                 clusters.remove(head);
741                 terms.addAll(extents.get(head));
742                 final List<Term> path = Lists.newArrayList();
743                 for (Dep dep = document.getDepToTerm(head); dep != null; dep = document
744                         .getDepToTerm(dep.getFrom())) {
745                     final Term term = dep.getFrom();
746                     path.add(term);
747                     if (heads.contains(term)) {
748                         terms.addAll(path);
749                         path.clear();
750                         spanHead = term;
751                     }
752                 }
753             }
754             List<Term> spanTerms = Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms);
755             if (canAddTerms) {
756                 final List<Term> docTerms = document.getTerms();
757                 spanTerms = Lists.newArrayList(docTerms.subList(
758                         docTerms.indexOf(spanTerms.get(0)),
759                         docTerms.indexOf(spanTerms.get(spanTerms.size() - 1)) + 1));
760             }
761             result.add(KAFDocument.newTermSpan(spanTerms, spanHead));
762         }
763         return result;
764     }

766     public static final List<Span<Term>> splitSpans(final KAFDocument document,
767             final Iterable<Span<Term>> spans) {
768 
769         // Identify all the heads taking coordination into consideration
770         final Set<Term> heads = Sets.newHashSet();
771         final Set<Term> terms = Sets.newHashSet();
772         for (final Span<Term> span : spans) {
773             final Term head = extractHead(document, span);
774             heads.add(head);
775             terms.addAll(span.getTargets());
776 
777             final List<Term> queue = Lists.newLinkedList();
778             queue.add(head);
779             while (!queue.isEmpty()) {
780                 final Term term = queue.remove(0);
781                 for (final Dep dep : document.getDepsFromTerm(term)) {
782                     final String func = dep.getRfunc();
783                     if ("COORD".equals(func) || "CONJ".equals(func)) {
784                         final Term t = dep.getTo();
785                         queue.add(t);
786                         if ("CC".equals(t.getMorphofeat())
787                                 || !Character.isLetter(t.getMorphofeat().charAt(0))) {
788                             heads.add(term);
789                         }
790                     }
791                 }
792             }
793         }
794 
795         // Build and return a span for each head
796         final Set<Term> excluded = document.getTermsByDepDescendants(heads);
797         final List<Span<Term>> result = Lists.newArrayList();
798         for (final Term head : heads) {
799             final Set<Term> extent = document.getTermsByDepAncestors(ImmutableSet.of(head));
800             extent.removeAll(excluded);
801             extent.add(head);
802             extent.retainAll(terms);
803             if (!extent.isEmpty()) {
804                 result.add(KAFDocument.newTermSpan(Ordering.from(Term.OFFSET_COMPARATOR)
805                         .sortedCopy(extent), head));
806             }
807         }
808         return result;
809     }

811     public static final List<Span<Term>> splitSpan(final KAFDocument document,
812             final Span<Term> span, final Iterable<Term> heads) {
813 
814         final Set<Term> excludedTerms = document.getTermsByDepDescendants(heads);
815         final List<Span<Term>> spans = Lists.newArrayList();
816         for (final Term head : heads) {
817             final Set<Term> terms = document.getTermsByDepAncestors(ImmutableSet.of(head));
818             terms.removeAll(excludedTerms);
819             terms.add(head);
820             terms.retainAll(span.getTargets());
821             if (!terms.isEmpty()) {
822                 spans.add(KAFDocument.newTermSpan(Ordering.from(Term.OFFSET_COMPARATOR)
823                         .sortedCopy(terms), head));
824             }
825         }
826         return spans;
827     }

880     public static String extendedPos(final KAFDocument document, final Term term) {
881         final String pos = term.getMorphofeat();
882         final String lemma = term.getLemma().toLowerCase();
883         if ("some".equals(lemma) || "many".equals(lemma) || "all".equals(lemma)
884                 || "few".equals(lemma) || "this".equals(lemma) || "these".equals(lemma)
885                 || "that".equals(lemma) || "those".equals(lemma)) {
886             final Dep dep = document.getDepToTerm(term);
887             if (dep == null || !"NMOD".equals(dep.getRfunc())) {
888                 return pos + "P"; // determiner (DT) or adj (JJ) used as demonstrative pronoun
889             }
890         }
891         return pos;
892     }

894     public static Boolean isActiveForm(final KAFDocument document, final Term term) {
895         final String word = term.getStr().toLowerCase();
896         final String pos = term.getMorphofeat();
897         if (!pos.startsWith("V")) {
898             return null;
899         }
900         if (word.equals("been") || !pos.equals("VBN")) {
901             return Boolean.TRUE;
902         }
903         return isActiveFormHelper(document, term);
904     }

906     private static Boolean isActiveFormHelper(final KAFDocument document, final Term term) {
907         final Dep dep = document.getDepToTerm(term);
908         if (dep == null) {
909             return Boolean.FALSE;
910         }
911         final Term parent = dep.getFrom();
912         final String word = parent.getStr().toLowerCase();
913         final String pos = parent.getMorphofeat();
914         if (pos.startsWith("NN")) {
915             return Boolean.FALSE;
916         }
917         if (word.matches("am|are|is|was|were|be|been|being")) {
918             return Boolean.FALSE;
919         }
920         if (word.matches("ha(ve|s|d|ving)")) {
921             return Boolean.TRUE;
922         }
923 
924         if (pos.matches("VBZ|VBD|VBP|MD")) {
925             return Boolean.FALSE;
926         }
927         return isActiveFormHelper(document, parent);
928     }

930     public static java.util.function.Predicate<Term> matchExtendedPos(final KAFDocument document,
931             final String... posPrefixes) {
932         return new java.util.function.Predicate<Term>() {

934             @Override

937                 for (final String prefix : posPrefixes) {

939                         return true;

943             }

948     // extracts descendents that are consecutive with the supplied head
949     public static Set<Term> getTermsByDepAncestor(final KAFDocument document, final Term head,
950             final boolean consecutive) {
951         final Set<Term> descendants = document.getTermsByDepAncestors(ImmutableSet.of(head));
952         if (consecutive) {
953             final List<Term> sortedTerms = Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(
954                     descendants);
955             final int[] indexes = new int[sortedTerms.size()];
956             for (int i = 0; i < sortedTerms.size(); ++i) {
957                 indexes[i] = document.getTerms().indexOf(sortedTerms.get(i));
958             }
959             final int h = sortedTerms.indexOf(head);
960             boolean filtered = false;
961             for (int i = h + 1; i < indexes.length; ++i) {
962                 filtered |= indexes[i] > indexes[i - 1] + 1;
963                 if (filtered) {
964                     descendants.remove(sortedTerms.get(i));
965                 }
966             }
967             filtered = false;
968             for (int i = h - 1; i >= 0; --i) {
969                 filtered |= indexes[i] < indexes[i + 1] - 1;
970                 if (filtered) {
971                     descendants.remove(sortedTerms.get(i));
972                 }
973             }
974         }
975         return descendants;
976     }
