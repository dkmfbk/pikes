4   import java.util.ArrayList;

11  public class Role implements Serializable {
12  
13      private List<Word> span = new ArrayList<>();
14      private String label;
15  
16      public Role(List<Word> span, String label) {
17          this.span = span;
18          this.label = label;
19      }
20  
21      public Role(Word word, String label) {
22          this.span.add(word);
23          this.label = label;
24      }
25  
26      public List<Word> getSpan() {
27          return span;
28      }
29  
30      public String getLabel() {
31          return label;
32      }
33  
34      @Override public String toString() {
35          return "Role{" +
36                  "span=" + span +
37                  ", label='" + label + '\'' +
38                  '}';
39      }
40  }
