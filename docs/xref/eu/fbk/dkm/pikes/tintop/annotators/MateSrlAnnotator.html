26  public class MateSrlAnnotator implements Annotator {
27  
28      private SemanticRoleLabeler labeler;
29      private SemanticRoleLabeler labelerBe = null;
30      int maxLen;
31  
32      public MateSrlAnnotator(String annotatorName, Properties props) {
33  
34          String model = props.getProperty(annotatorName + ".model", Defaults.MATE_MODEL);
35          maxLen = Defaults.getInteger(props.getProperty(annotatorName + ".maxlen"), Defaults.MAXLEN);
36          String modelBe = props.getProperty(annotatorName + ".model_be", Defaults.MATE_MODEL_BE);
37  
38          labeler = MateSrlModel.getInstance(new File(model)).getLabeler();
39  
40          if (modelBe != null) {
41              labelerBe = MateSrlBeModel.getInstance(new File(modelBe)).getLabeler();
42          }
43      }
44  
45      public static Sentence createMateSentence(CoreMap stanfordSentence) {
46          Sentence ret;
47  
48          java.util.List<CoreLabel> get = stanfordSentence.get(CoreAnnotations.TokensAnnotation.class);
49          int size = get.size();
50  
51          String[] forms = new String[size + 1];
52          String[] poss = new String[size + 1];
53          String[] lemmas = new String[size + 1];
54          String[] feats = new String[size + 1];
55          String[] labels = new String[size];
56          int[] parents = new int[size];
57  
58          forms[0] = "<root>";
59          poss[0] = "<root>";
60          lemmas[0] = "<root>";
61          feats[0] = "<root>";
62  
63          for (int i = 0; i < get.size(); i++) {
64              CoreLabel token = get.get(i);
65              forms[i + 1] = token.get(CoreAnnotations.TextAnnotation.class);
66              poss[i + 1] = token.get(CoreAnnotations.PartOfSpeechAnnotation.class);
67              lemmas[i + 1] = token.get(CoreAnnotations.LemmaAnnotation.class);
68              feats[i + 1] = null;
69  
70              labels[i] = token.get(CoreAnnotations.CoNLLDepTypeAnnotation.class);
71              parents[i] = token.get(CoreAnnotations.CoNLLDepParentIndexAnnotation.class) + 1;
72          }
73  
74          ret = new Sentence(forms, lemmas, poss, feats);
75  
76          ret.setHeadsAndDeprels(parents, labels);
77  
78          return ret;
79      }
80  
81      @Override
82      public void annotate(Annotation annotation) {
83          if (annotation.has(CoreAnnotations.SentencesAnnotation.class)) {
84              for (CoreMap stanfordSentence : annotation.get(CoreAnnotations.SentencesAnnotation.class)) {
85  
86                  Sentence sentence;
87                  try {
88                      sentence = createMateSentence(stanfordSentence);
89                  } catch (Exception e) {
90                      // NullPointerException
91                      continue;
92                  }
93  
94                  labeler.parseSentence(sentence);
95  
96                  for (Word word : sentence) {
97                      int tokenID = word.getIdx() - 1;
98                      if (tokenID < 0) {
99                          continue;
100                     }
101                     try {
102                         stanfordSentence.get(CoreAnnotations.TokensAnnotation.class).get(tokenID)
103                                 .set(PikesAnnotations.MateTokenAnnotation.class, word);
104                     } catch (Exception e) {
105                         e.printStackTrace();
106                     }
107 
108                 }
109 
110                 for (Predicate predicate : sentence.getPredicates()) {
111                     int tokenID = predicate.getIdx() - 1;
112                     try {
113                         stanfordSentence.get(CoreAnnotations.TokensAnnotation.class).get(tokenID)
114                                 .set(PikesAnnotations.MateAnnotation.class, predicate);
115                     } catch (Exception e) {
116                         e.printStackTrace();
117                     }
118                 }
119 
120                 if (labelerBe != null) {
121                     Sentence sentenceBe = createMateSentence(stanfordSentence);
122                     labelerBe.parseSentence(sentenceBe);
123 
124                     for (Predicate predicate : sentenceBe.getPredicates()) {
125                         int tokenID = predicate.getIdx() - 1;
126                         String lemma = stanfordSentence.get(CoreAnnotations.TokensAnnotation.class).get(tokenID).get(
127                                 CoreAnnotations.LemmaAnnotation.class);
128                         if (lemma.equals("be")) {
129                             try {
130                                 stanfordSentence.get(CoreAnnotations.TokensAnnotation.class).get(tokenID)
131                                         .set(PikesAnnotations.MateAnnotation.class, predicate);
132                             } catch (Exception e) {
133                                 e.printStackTrace();
134                             }
135                         }
136                     }
137                 }
138             }
139         } else {
140             throw new RuntimeException("unable to find words/tokens in: " + annotation);
141         }
142 
143     }
