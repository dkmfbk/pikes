14  import java.util.List;
15  import java.util.Properties;
16  import java.util.Set;

19   * Created by alessio on 06/05/15.

21  
22  public class AnnaPosAnnotator implements Annotator {
23  
24  	private Tagger tagger;
25  
26  	public AnnaPosAnnotator(String annotatorName, Properties props) {
27  		File posModel = new File(props.getProperty(annotatorName + ".model", Defaults.ANNA_POS_MODEL));
28  		tagger = AnnaPosModel.getInstance(posModel).getTagger();
29  	}
30  
31  	@Override
32  	public void annotate(Annotation annotation) {
33  		if (annotation.has(CoreAnnotations.SentencesAnnotation.class)) {
34  			for (CoreMap sentence : annotation.get(CoreAnnotations.SentencesAnnotation.class)) {
35  				List<CoreLabel> tokens = sentence.get(CoreAnnotations.TokensAnnotation.class);
36  
37  				String[] annaTokens = new String[tokens.size() + 1];
38  				annaTokens[0] = "<ROOT>";
39  
40  				for (int i = 0, sz = tokens.size(); i < sz; i++) {
41  					CoreLabel thisToken = tokens.get(i);
42  					annaTokens[i + 1] = thisToken.originalText();
43  				}
44  
45  				SentenceData09 instance = new SentenceData09();
46  				instance.init(annaTokens);
47  				tagger.apply(instance);
48  
49  				for (int i = 0, sz = tokens.size(); i < sz; i++) {
50  					CoreLabel thisToken = tokens.get(i);
51  					String pos = AnnotatorUtils.parenthesisToCode(instance.ppos[i + 1]);
52  					thisToken.set(CoreAnnotations.PartOfSpeechAnnotation.class, pos);
53  				}
54  			}
55  		}
56  		else {
57  			throw new RuntimeException("unable to find words/tokens in: " + annotation);
58  		}
59  
60  	}
61  
62  	@Override
63  	public Set<Requirement> requirementsSatisfied() {
64  		return Collections.singleton(POS_REQUIREMENT);
65  	}
66  
67  	@Override
68  	public Set<Requirement> requires() {
69  		return TOKENIZE_AND_SSPLIT;
70  	}
71  }
