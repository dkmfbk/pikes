1   package eu.fbk.dkm.pikes.tintop.annotators;

8   import edu.stanford.nlp.util.CoreMap;
9   import eu.fbk.fcw.ukb.UKBModel;
10  import eu.fbk.fcw.ukb.UKB_MT;
11  import eu.fbk.utils.core.PropertiesUtils;
12  
13  import java.io.IOException;
14  import java.util.*;
15

17   * Created by alessio on 06/05/15.

19  
20  public class UKBAnnotator implements Annotator {
21  
22  	private UKB_MT tagger;
23  	int maxLen;
24  
25  	public UKBAnnotator(String annotatorName, Properties props) {
26  		Properties newProps = PropertiesUtils.dotConvertedProperties(props, annotatorName);
27  		maxLen = Defaults.getInteger(newProps.getProperty("maxlen"), Defaults.MAXLEN);
28  		try {
29  			tagger = UKBModel.getInstance(newProps).getTagger();
30  		} catch (IOException e) {
31  			e.printStackTrace();
32  		}
33  	}
34  
35  	@Override
36  	public void annotate(Annotation annotation) {
37  		if (annotation.has(CoreAnnotations.SentencesAnnotation.class)) {
38  			for (CoreMap sentence : annotation.get(CoreAnnotations.SentencesAnnotation.class)) {
39  				List<CoreLabel> tokens = sentence.get(CoreAnnotations.TokensAnnotation.class);
40  				if (maxLen > 0 && tokens.size() > maxLen) {
41  					continue;
42  				}
43  
44  				ArrayList<HashMap<String, String>> terms = new ArrayList<>();
45  				for (CoreLabel token : tokens) {
46  					HashMap<String, String> term = new HashMap<>();
47  
48  					term.put("simple_pos", token.get(PikesAnnotations.SimplePosAnnotation.class));
49  					term.put("lemma", token.get(CoreAnnotations.LemmaAnnotation.class));
50  
51  					terms.add(term);
52  				}
53  
54  				try {
55  					tagger.run(terms);
56  				} catch (IOException e) {
57  					e.printStackTrace();
58  				}
59  
60  				for (int i = 0, sz = tokens.size(); i < sz; i++) {
61  					CoreLabel thisToken = tokens.get(i);
62  					String wn = terms.get(i).get("wordnet");
63  					thisToken.set(PikesAnnotations.UKBAnnotation.class, wn);
64  				}
65  			}
66  		}
67  		else {
68  			throw new RuntimeException("unable to find words/tokens in: " + annotation);
69  		}
70  	}
71  
72  	@Override
73  	public Set<Requirement> requirementsSatisfied() {
74  		return Collections.singleton(PikesAnnotations.WORDNET_REQUIREMENT);
75  	}
76  
77  	@Override
78  	public Set<Requirement> requires() {
79  		return Collections.unmodifiableSet(new ArraySet<Requirement>(TOKENIZE_REQUIREMENT, SSPLIT_REQUIREMENT, LEMMA_REQUIREMENT, PikesAnnotations.SIMPLEPOS_REQUIREMENT));
80  	}
81  }
