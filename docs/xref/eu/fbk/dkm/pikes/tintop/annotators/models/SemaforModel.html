11  public class SemaforModel {

13      private static SemaforModel instance;
14      private Semafor parser;
15      private static final Logger LOGGER = LoggerFactory.getLogger(SemaforModel.class);
16  
17      private SemaforModel(String modelDir) {
18          LOGGER.info("Loading model for Semafor");
19          try {
20              parser = Semafor.getSemaforInstance(modelDir);
21          } catch (Exception e) {
22              e.printStackTrace();
23          }
24      }
25  
26      public static SemaforModel getInstance(String modelDir) {
27          if (instance == null) {
28              instance = new SemaforModel(modelDir);
29          }
30  
31          return instance;
32      }
33  
34      public Semafor getParser() {
35          return parser;
36      }
