62      @Override
63      public List<LinkingTag> tag(String text) throws Exception {
64  
65          ArrayList<LinkingTag> ret = new ArrayList<>();
66          Map<String, String> pars;
67  
68          pars = new HashMap<>();
69          pars.put("min_weight", minWeight.toString());
70          pars.put("disambiguation", "1");
71          pars.put("topic", "1");
72          pars.put("include_text", "0");
73          pars.put("image", "1");
74          pars.put("class", "1");
75          pars.put("app_id", "0");
76          pars.put("app_key", "0");
77          pars.put("text", text);
78  
79          LOGGER.debug("Text length: {}", text.length());
80          LOGGER.debug("Pars: {}", pars);
81  
82          Map<String, Object> userData;
83          String output = request(pars);
84  
85          ObjectMapper mapper = new ObjectMapper();
86          userData = mapper.readValue(output, Map.class);
87  
88          LinkedHashMap annotation = (LinkedHashMap) userData.get(new String("annotation"));
89          if (annotation != null) {
90              String lang = annotation.get("lang").toString();
91              String language = (lang == null || lang.equals("en")) ? "" : lang + ".";
92              ArrayList<LinkedHashMap> keywords = (ArrayList<LinkedHashMap>) annotation.get(new String("keyword"));
93              if (keywords != null) {
94                  for (LinkedHashMap keyword : keywords) {
95                      LinkedHashMap sense = (LinkedHashMap) keyword.get("sense");
96                      ArrayList dbpClass = (ArrayList) keyword.get("class");
97                      ArrayList<LinkedHashMap> images = (ArrayList<LinkedHashMap>) keyword.get("image");
98                      ArrayList<LinkedHashMap> ngrams = (ArrayList<LinkedHashMap>) keyword.get("ngram");
99                      for (LinkedHashMap ngram : ngrams) {
100                         String originalText = (String) ngram.get("form");
101                         LinkedHashMap span = (LinkedHashMap) ngram.get("span");
102 
103                         Integer start = (Integer) span.get("start");
104                         Integer end = (Integer) span.get("end");
105 
106                         LinkingTag tag = new LinkingTag(
107                                 start,
108                                 String.format("http://" + language + "dbpedia.org/resource/%s",
109                                         (String) sense.get("page")),
110                                 Double.parseDouble(keyword.get("rel").toString()),
111                                 originalText,
112                                 end - start,
113                                 LABEL
114                         );
115 
116                         //todo: add to conf
117                         if (images != null && images.size() > 0) {
118                             try {
119                                 tag.setImage(images.get(0).get("image").toString());
120                             } catch (Exception e) {
121                                 // ignored
122                             }
123                         }
124 
125                         if (extractTypes) {
126                             tag.addTypesFromML(dbpClass);
127                         }
128                         ret.add(tag);
129                     }
130                 }
131             }
132         }
133 
134         return ret;
135     }
