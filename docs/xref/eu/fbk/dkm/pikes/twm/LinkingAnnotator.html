19  public class LinkingAnnotator implements Annotator {
20  
21      Linking tagger;
22      public static final String DBPS_ANNOTATOR = "dbpedia-candidates";
23  
24      private static HashMap<String, Class<? extends Linking>> annotators = new HashMap<>();
25  
26      static {
27          annotators.put("dbpedia-candidates", DBpediaSpotlightCandidates.class);
28          annotators.put("dbpedia-annotate", DBpediaSpotlightAnnotate.class);
29          annotators.put("ml-annotate", MachineLinking.class);
30      }
31  
32      public LinkingAnnotator(String annotatorName, Properties props) throws Exception {
33          Properties newProps = PropertiesUtils.dotConvertedProperties(props, annotatorName);
34  
35          String annotator = newProps.getProperty("annotator", DBPS_ANNOTATOR);
36          Class<? extends Linking> myClass = annotators.get(annotator);
37          Constructor<? extends Linking> myConstructor = myClass.getConstructor(Properties.class);
38          tagger = myConstructor.newInstance(newProps);
39      }
40  
41      @Override
42      public void annotate(Annotation annotation) {
43          String text = annotation.get(CoreAnnotations.TextAnnotation.class);
44          if (text == null) {
45              throw new RuntimeException("Text is null");
46          }
47  
48          List<LinkingTag> tags = new ArrayList<>();
49          try {
50              tags = tagger.tag(text);
51          } catch (EOFException e) {
52              e.printStackTrace();
53          } catch (Exception e) {
54              // ignore
55  //            throw new RuntimeException(e);
56          }
57  
58          List<LinkingTag> annotatedEntities = annotation.get(TWMAnnotations.LinkingAnnotations.class);
59          List<LinkingTag> entities = new ArrayList<>();
60          if (annotatedEntities != null) {
61              entities.addAll(annotatedEntities);
62          }
63  
64          HashMap<Integer, LinkingTag> index = new HashMap<>();
65          for (LinkingTag tag : tags) {
66              entities.add(tag);
67              int end = tag.getOffset() + tag.getLength();
68              for (int i = tag.getOffset(); i < end; i++) {
69                  index.put(i, tag);
70              }
71          }
72  
73          annotation.set(TWMAnnotations.LinkingAnnotations.class, entities);
74  
75          if (annotation.has(CoreAnnotations.SentencesAnnotation.class)) {
76              for (CoreMap sentence : annotation.get(CoreAnnotations.SentencesAnnotation.class)) {
77                  List<CoreLabel> tokens = sentence.get(CoreAnnotations.TokensAnnotation.class);
78  
79                  for (CoreLabel token : tokens) {
80                      LinkingTag startEntity = index.get(token.beginPosition());
81                      if (startEntity == null) {
82                          continue;
83                      }
84  
85                      LinkingTag endEntity = index.get(token.endPosition() - 1);
86                      if (endEntity == null) {
87                          continue;
88                      }
89  
90                      if (startEntity.equals(endEntity)) {
91                          token.set(TWMAnnotations.DBpediaSpotlightAnnotation.class, startEntity);
92                      }
93                  }
94              }
95          } else {
96              throw new RuntimeException("unable to find words/tokens in: " + annotation);
97          }
98      }
99  
100     @Override
101     public Set<Requirement> requirementsSatisfied() {
102         return Collections.unmodifiableSet(
103                 new ArraySet<Requirement>(TWMAnnotations.DBPS_REQUIREMENT, TWMAnnotations.LINKING_REQUIREMENT));
104     }
105 
106     @Override
107     public Set<Requirement> requires() {
108         return Annotator.TOKENIZE_AND_SSPLIT;
109     }
110 }
