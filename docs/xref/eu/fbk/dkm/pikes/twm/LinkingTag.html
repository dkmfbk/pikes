72      public void addTypesFromML(ArrayList classes) {
73          for (Object aClass : classes) {
74              LinkedHashMap map = (LinkedHashMap) aClass;
75              String type = (String) map.get("label");
76              type = type.replaceAll("\\s+", "");
77              addType(Category.DBPEDIA, type);
78          }
79      }

81      public void addTypesFromDBpedia(String types) {
82          String[] parts = types.split(",");
83          for (String catType : parts) {
84              catType = catType.trim();
85              String[] typeParts = catType.split(":");
86              if (typeParts.length < 2) {
87                  continue;
88              }
89  
90              String cat = typeParts[0].trim();
91              String type = typeParts[1].trim();
92  
93              switch (cat.toLowerCase()) {
94              case "dbpedia":
95                  addType(Category.DBPEDIA, type);
96                  break;
97              case "schema":
98                  addType(Category.SCHEMA, type);
99              }
100         }
101     }

103     public HashMap<String, HashSet<String>> getStringTypes() {
104         HashMap<String, HashSet<String>> ret = new HashMap<>();
105         for (Category category : types.keySet()) {
106             String catString = category.toString();
107             ret.put(catString, new HashSet<>());
108             ret.get(catString).addAll(types.get(category));
109         }

112     }
