1   package eu.fbk.dkm.pikes.depparseannotation;
2   
3   import edu.stanford.nlp.ling.IndexedWord;
4   import edu.stanford.nlp.semgraph.SemanticGraph;
5   import edu.stanford.nlp.semgraph.SemanticGraphEdge;
6   import edu.stanford.nlp.util.Generics;
7   
8   import java.util.*;
9   
10  /**
11   * Created by alessio on 10/11/15.
12   */
13  
14  public class DepParseInfo {
15  
16      private HashMap<Integer, Integer> depParents = new HashMap<>();
17      private HashMap<Integer, String> depLabels = new HashMap<>();
18  
19      private static String space(int width) {
20          StringBuilder b = new StringBuilder();
21          for (int i = 0; i < width; i++) {
22              b.append(' ');
23          }
24          return b.toString();
25      }
26  
27      // helper for toString()
28      private static void recToString(IndexedWord curr, StringBuilder sb, int offset, Set<IndexedWord> used,
29              SemanticGraph dependencies, HashMap<Integer, String> depLabels, HashMap<Integer, Integer> depParents) {
30          used.add(curr);
31          List<SemanticGraphEdge> edges = dependencies.outgoingEdgeList(curr);
32          Collections.sort(edges);
33          for (SemanticGraphEdge edge : edges) {
34              IndexedWord target = edge.getTarget();
35  
36              depParents.put(target.index(), edge.getSource().index());
37              depLabels.put(target.index(), edge.getRelation().toString());
38  
39              sb.append(space(2 * offset)).append("-> ").append(target).append(" (").append(edge.getRelation())
40                      .append(")\n");
41  
42              if (!used.contains(target)) { // recurse
43                  recToString(target, sb, offset + 1, used, dependencies, depLabels, depParents);
44              }
45          }
46      }
47  
48      public DepParseInfo(HashMap<Integer, Integer> depParents,
49              HashMap<Integer, String> depLabels) {
50          this.depParents = depParents;
51          this.depLabels = depLabels;
52      }
53  
54      public DepParseInfo(SemanticGraph dependencies) {
55          Collection<IndexedWord> rootNodes = dependencies.getRoots();
56          if (rootNodes.isEmpty()) {
57              // Shouldn't happen, but return something!
58              return;
59          }
60  
61          StringBuilder sb = new StringBuilder();
62          Set<IndexedWord> used = Generics.newHashSet();
63          for (IndexedWord root : rootNodes) {
64              depParents.put(root.index(), 0);
65              depLabels.put(root.index(), "root");
66              sb.append("-> ").append(root).append(" (root)\n");
67              recToString(root, sb, 1, used, dependencies, depLabels, depParents);
68          }
69  
70          Set<IndexedWord> nodes = Generics.newHashSet(dependencies.vertexSet());
71          nodes.removeAll(used);
72          while (!nodes.isEmpty()) {
73              IndexedWord node = nodes.iterator().next();
74              sb.append(node).append("\n");
75              recToString(node, sb, 1, used, dependencies, depLabels, depParents);
76              nodes.removeAll(used);
77          }
78  
79      }
80  
81      public HashMap<Integer, Integer> getDepParents() {
82          return depParents;
83      }
84  
85      public HashMap<Integer, String> getDepLabels() {
86          return depLabels;
87      }
88  
89      @Override public String toString() {
90          return "DepParseInfo{" +
91                  "depParents=" + depParents +
92                  ", depLabels=" + depLabels +
93                  '}';
94      }
95  }
