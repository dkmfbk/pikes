26          SAXBuilder builder = new SAXBuilder();

50      /**
51       * Writes the content of a given KAFDocument to a file.
52       */
53      static void save(KAFDocument kaf, File file) {
54          try {
55              Writer out = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(file), "UTF8"));
56              out.write(kafToStr(kaf));
57              out.flush();
58              out.close();
59          } catch (Exception e) {
60              System.err.println(String.format("Error in writing file %s: %s", file.getAbsolutePath(), e.getMessage()));
61          }
62      }

77      /**
78       * Returns a string containing the XML content of a KAFDocument object.
79       */
80      static String kafToStr(KAFDocument kaf) {
81          XMLOutputter out = new XMLOutputter(Format.getPrettyFormat().setLineSeparator(LineSeparator.UNIX));
82  //		out.getFormat().setTextMode(Format.TextMode.PRESERVE);
83          Document jdom = KAFToDOM(kaf);
84          return out.outputString(jdom);
85      }

991     private static void DOMToTerm(Element termElem, KAFDocument kaf, boolean isComponent, Map<String, WF> wfIndex,
992             Map<String, Term> termIndex) throws KAFNotValidException {
993         String tid;
994         try {
995             tid = getAttribute("id", termElem);
996         } catch (Exception e) {
997             tid = getAttribute("tid", termElem);
998         }
999         Element spanElem = termElem.getChild("span");
1000         if (spanElem == null) {
1001             throw new IllegalStateException("Every term must contain a span element");
1002         }

1004         Span<WF> span = kaf.newWFSpan();

1008             WF wf = wfIndex.get(wfId);
1009             if (wf == null) {
1010                 throw new KAFNotValidException("Wf " + wfId + " not found when loading term " + tid);
1011             }

1013         }
1014         Term newTerm = kaf.newTerm(tid, span, isComponent);
1015         String type = getOptAttribute("type", termElem);
1016         if (type != null) {
1017             newTerm.setType(type);
1018         }
1019         String lemma = getOptAttribute("lemma", termElem);
1020         if (lemma != null) {
1021             newTerm.setLemma(lemma);
1022         }
1023         String supersenseTag = getOptAttribute("supersense", termElem);
1024         if (supersenseTag != null) {
1025             newTerm.setSupersenseTag(supersenseTag);
1026         }
1027         String wordnetSense = getOptAttribute("wordnet", termElem);
1028         if (wordnetSense != null) {
1029             newTerm.setWordnetSense(wordnetSense);
1030         }
1031         String bbnTag = getOptAttribute("bbn", termElem);
1032         if (bbnTag != null) {
1033             newTerm.setBBNTag(bbnTag);
1034         }
1035         String pos = getOptAttribute("pos", termElem);
1036         if (pos != null) {
1037             newTerm.setPos(pos);
1038         }
1039         String tMorphofeat = getOptAttribute("morphofeat", termElem);
1040         if (tMorphofeat != null) {
1041             newTerm.setMorphofeat(tMorphofeat);
1042         }
1043         String tHead = getOptAttribute("head", termElem);
1044         String termcase = getOptAttribute("case", termElem);
1045         if (termcase != null) {
1046             newTerm.setCase(termcase);
1047         }
1048         List<Element> sentimentElems = termElem.getChildren("sentiment");
1049         if (sentimentElems.size() > 0) {
1050             Element sentimentElem = sentimentElems.get(0);
1051             Term.Sentiment newSentiment = kaf.newSentiment();
1052             String sentResource = getOptAttribute("resource", sentimentElem);
1053             if (sentResource != null) {
1054                 newSentiment.setResource(sentResource);
1055             }
1056             String sentPolarity = getOptAttribute("polarity", sentimentElem);
1057             if (sentPolarity != null) {
1058                 newSentiment.setPolarity(sentPolarity);
1059             }
1060             String sentStrength = getOptAttribute("strength", sentimentElem);
1061             if (sentStrength != null) {
1062                 newSentiment.setStrength(sentStrength);
1063             }
1064             String sentSubjectivity = getOptAttribute("subjectivity", sentimentElem);
1065             if (sentSubjectivity != null) {
1066                 newSentiment.setSubjectivity(sentSubjectivity);
1067             }
1068             String sentSentimentSemanticType = getOptAttribute("sentiment_semantic_type", sentimentElem);
1069             if (sentSentimentSemanticType != null) {
1070                 newSentiment.setSentimentSemanticType(sentSentimentSemanticType);
1071             }
1072             String sentSentimentModifier = getOptAttribute("sentiment_modifier", sentimentElem);
1073             if (sentSentimentModifier != null) {
1074                 newSentiment.setSentimentModifier(sentSentimentModifier);
1075             }
1076             String sentSentimentMarker = getOptAttribute("sentiment_marker", sentimentElem);
1077             if (sentSentimentMarker != null) {
1078                 newSentiment.setSentimentMarker(sentSentimentMarker);
1079             }
1080             String sentSentimentProductFeature = getOptAttribute("sentiment_product_feature", sentimentElem);
1081             if (sentSentimentProductFeature != null) {
1082                 newSentiment.setSentimentProductFeature(sentSentimentProductFeature);
1083             }
1084             newTerm.setSentiment(newSentiment);
1085         }
1086         if (!isComponent) {
1087             List<Element> termsComponentElems = termElem.getChildren("component");
1088             for (Element termsComponentElem : termsComponentElems) {
1089                 DOMToTerm(termsComponentElem, kaf, true, wfIndex, termIndex);
1090             }
1091         }
1092         List<Element> externalReferencesElems = termElem.getChildren("externalReferences");
1093         if (externalReferencesElems.size() > 0) {
1094             List<ExternalRef> externalRefs = getExternalReferences(externalReferencesElems.get(0), kaf);
1095             newTerm.addExternalRefs(externalRefs);
1096         }
1097         termIndex.put(newTerm.getId(), newTerm);
1098     }

1100     private static Span<Term> loadTermSpan(Element spanElem, HashMap<String, Term> terms, String objId)
1101             throws KAFNotValidException {
1102         List<Element> targetElems = spanElem.getChildren("target");
1103         if (targetElems.size() < 1) {
1104             throw new KAFNotValidException("A span element can not be empty");
1105         }
1106         Span<Term> span = KAFDocument.newTermSpan();
1107         for (Element targetElem : targetElems) {
1108             String targetId = getAttribute("id", targetElem);
1109             boolean isHead = isHead(targetElem);
1110             Term targetTerm = terms.get(targetId);
1111             if (targetTerm == null) {
1112                 throw new KAFNotValidException("Term object " + targetId + " not found when loading object " + objId);
1113             }
1114             span.addTarget(targetTerm, isHead);
1115         }
1116         return span;
1117     }
